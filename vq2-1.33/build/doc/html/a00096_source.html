76  template<typename GRAPH,
77  typename SIMILARITY,
78  typename VECTOR_OP,
79  typename SPLIT,
80  typename SET_ITER>
81  void process(GRAPH& g,
82  unsigned int k,
83  SIMILARITY& sim,
84  VECTOR_OP& op,
85  SPLIT& split,
86  const SET_ITER& begin,const SET_ITER& end,
87  double biggest_similarity) {
88  unsigned int size = (unsigned int)(end-begin);
89  std::vector<typename GRAPH::vertex_type*> nodes;
90  std::vector<int> labels;
91  std::vector<int> nb;
92  SET_ITER i;
93  double dummy;
94  bool achieved;
95  typename std::vector<int>::iterator liter,lend;
96  typename std::vector<typename GRAPH::vertex_type*>::iterator piter,pend;
97  typename GRAPH::ref_vertex_type clst;
98 
99  g.clear();
100 
101  if(k == 0 || size == 0 || size <= k)
102  return;
103 
104  labels.resize(size);
105 
106  // We create a prototype, that is the barycenter of the data.
107  typename GRAPH::vertex_value_type::prototype_type first_proto;
108  op.raz(first_proto);
109  typename GRAPH::ref_vertex_type first_node_ref(g += unit_of(first_proto));
110  typename GRAPH::vertex_value_type::prototype_type& first_proto_ref = (*first_node_ref).value.prototype();
111  for(i=begin; i!= end; ++i)
112  op.add(first_proto_ref,*i);
113  op.div(first_proto_ref,(double)size);
114 
115  // We add the node in nodes.
116  {
117  typename GRAPH::vertex_type& n = *first_node_ref;
118  n.stuff.tag = (int)(nodes.size());
119  nodes.push_back(&n);
120  }
121 
122  // Check if we have enough nodes.
123  int remaining = k - nodes.size();
124  while(remaining > 0) {
125  // we need to split existing nodes.
126  int nb_splits = std::min(nodes.size(),(typename std::vector<typename GRAPH::vertex_type*>::size_type)remaining);
127 
128  // Let us shuffle nodes, since we will systematically split
129  // from the first index.
130  // std::random_shuffle(nodes.begin(), nodes.end()); // quite useless
131 
132  // Let us now split the nb_split first nodes.
133  for(int splt=0; splt < nb_splits; ++splt) {
134  typename GRAPH::vertex_value_type::prototype_type proto;
135  split(proto,nodes[splt]->value.prototype());
136  typename GRAPH::ref_vertex_type node_ref(g += unit_of(proto));
137 
138  // We add the node in nodes.
139  typename GRAPH::vertex_type& n = *node_ref;
140  n.stuff.tag = (int)(nodes.size());
141  nodes.push_back(&n);
142  }
143 
144  // Now, we can start the k-mean labelling and updating
145  // procedure with the current nodes.
146 
147  nb.resize(nodes.size());
148  labels[0] = -1; // First labelling will at least change this value...
149  achieved = false;
150  while(!achieved) {
151  achieved = true;
152  std::fill(nb.begin(),nb.end(),0);
153 
154  // Compute labels for each sample.
155  for(i = begin, liter = labels.begin(), lend = labels.end();
156  liter != lend;
157  ++liter, ++i) {
158  clst = closest(g,sim,
159  biggest_similarity,
160  *i,dummy);
161  int l = (*clst).stuff.tag;
162  if(l != *liter) {
163  *liter = l;
164  achieved = false;
165  }
166  nb[l]++;
167  }
168 
169  // Le us update barycenters.
170  if(!achieved) {
171  for(piter = nodes.begin(), pend = nodes.end();
172  piter != pend;
173  ++piter)
174  op.raz((*piter)->value.prototype());
175 
176  for(liter = labels.begin(), lend = labels.end(), i=begin;
177  liter != lend;
178  ++liter,++i)
179  op.add((nodes[*liter])->value.prototype(),*i);
180 
181  std::vector<int>::iterator n;
182  for(piter = nodes.begin(), pend = nodes.end(), n = nb.begin();
183  piter != pend;
184  ++piter,++n)
185  if(*n != 0)
186  op.div((*piter)->value.prototype(), *n);
187  else
188  std::cout << "Warning : vq2::algo::kmeans::LBG::process : non winning prototype found." << std::endl;
189  }
190  }
191 
192  // Check if we have enough nodes.
193  remaining = k - nodes.size();
194  }
195  }
196  }
197  }
