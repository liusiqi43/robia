45  class Evolution {
46  public:
47 
48  std::vector<double> disto_distrib;
49  double min,max;
50  bool first_run;
51  EVOLUTION_PARAM& param;
52 
53  Evolution(EVOLUTION_PARAM& evolution_param)
54  : first_run(true),
55  param(evolution_param){}
56 
57  void clear(void) {
58  disto_distrib.clear();
59  }
60  void operator+=(double value) {disto_distrib.push_back(value);}
61  int operator()() {
62  int res = 0;
63  double a,b;
64  vq2::proba::shortest_confidence_interval(disto_distrib.begin(),
65  disto_distrib.end(),
66  [](double x) -> double {return x;},
67  param.delta(),a,b);
68  if(first_run) {
69  min = a;
70  max = b;
71  first_run = false;
72  }
73  else {
74  min += param.lowPassCoef()*(a-min);
75  max += param.lowPassCoef()*(b-max);
76  }
77 
78  double width;
79 
80  width = (max-min);
81  double _min = min + param.margin()*width;
82  double _max = min + (1-param.margin())*width;
83 
84  double nt = param.nbSamples()*param.target();
85  if(nt < _min)
86  res = 1;
87  else if(nt > _max)
88  res = -1;
89  return res;
90  }
91  };

321  template<typename GRAPH,typename GNGT_PARAMS,
322  typename SIMILARITY, typename LEARN>
323  static void submit(GNGT_PARAMS& params,
324  GRAPH& g,
325  SIMILARITY& distance,
326  LEARN& learn,
327  const typename LEARN::sample_type& xi,
328  bool dynamic_topology) {
329 
330  // Create two first vertextes from xi.
331  if(g.nbVertices() < 2) {
332  if(dynamic_topology) {
333  typename GRAPH::ref_vertex_type ref(g += typename GRAPH::vertex_value_type(xi));
334  (*ref).value.n = 1;
335  }
336  return;
337  }
338 
339  // Find the two closest vertices.
340  std::pair<double,double> closest_dists;
341  std::pair<typename GRAPH::ref_vertex_type,
342  typename GRAPH::ref_vertex_type> two_closest = vq2::algo::twoClosest(g,distance,
343  params.infinity(),
344  xi,closest_dists);
345 
346  if(dynamic_topology) {
347  // Update edges.
348  internal::UpdateAge<typename GRAPH::vertex_type,
349  typename GRAPH::ref_vertex_type,
350  typename GRAPH::edge_type,
351  GNGT_PARAMS> update_age(two_closest.second,params);
352  (*(two_closest.first)).for_each_edge(update_age);
353  if(!(update_age.found))
354  g.connect(0,two_closest.first,two_closest.second);
355  }
356 
357  // Update closest.
358  typename GRAPH::vertex_value_type& n1 = (*(two_closest.first)).value;
359  n1.e += closest_dists.first;
360  ++n1.n;
361  learn(params.learningRate(),n1,xi);
362 
363  // Update its neighbours.
364  internal::UpdateNeighbour<typename GRAPH::vertex_type,
365  typename GRAPH::ref_vertex_type,
366  typename GRAPH::edge_type,
367  LEARN,
368  GNGT_PARAMS> update_neighbour(two_closest.first,
369  xi,params,learn);
370  (*(two_closest.first)).for_each_edge(update_neighbour);
371  }

378  template<typename GNGT_PARAMS,
379  typename GRAPH,
380  typename LEARN,
381  typename GNGT_EVOLUTION>
382  void close_epoch(GNGT_PARAMS& params,
383  GRAPH& g,
384  LEARN& learn,
385  GNGT_EVOLUTION& stats,
386  bool dynamic_topology) {
387  unsigned int nb_vertices;
388  internal::MakeNeuronStat<typename GRAPH::ref_vertex_type,
389  typename GRAPH::vertex_value_type,
390  GNGT_EVOLUTION> make_neuron_stat(stats,nb_vertices);
391 
392 
393  if(!dynamic_topology)
394  return;
395 
396  g.for_each_vertex_ref(make_neuron_stat);
397  if(nb_vertices == 0)
398  return;
399 
400  int increment = stats();
401  if(increment == 0)
402  return;
403 
404  // T is too low, we have to remove some neuron.
405  if(increment < 0) {
406  make_neuron_stat.min.take();
407  make_neuron_stat.min.free();
408  return;
409  }
410 
411  // T is too high, we have to add some neuron.
412 
413  // Let us find the neighbour with the highest e
414  internal::FindHighestNeighbour<typename GRAPH::vertex_type,
415  typename GRAPH::ref_vertex_type,
416  typename GRAPH::edge_type,
417  GNGT_PARAMS> find_highest_neighbour(make_neuron_stat.max);
418  (*(make_neuron_stat.max)).for_each_edge(find_highest_neighbour);
419 
420  if(!(find_highest_neighbour.n2)) {
421  // max is an isolated node. We clone it.
422  typename GRAPH::ref_vertex_type clone(g += (*(make_neuron_stat.max)).value);
423  g.connect(0,make_neuron_stat.max,clone);
424  }
425  else {
426  typename GRAPH::vertex_value_type c;
427  typename GRAPH::vertex_value_type& a = (*(find_highest_neighbour.n1)).value;
428  typename GRAPH::vertex_value_type& b = (*(find_highest_neighbour.n2)).value;
429  c.prototype() = a.prototype();
430  learn(params.lambda(),c,b.prototype());
431  typename GRAPH::ref_vertex_type middle(g += c);
432  g.connect(0,find_highest_neighbour.n1,middle);
433  g.connect(0,find_highest_neighbour.n2,middle);
434  }
435  }
