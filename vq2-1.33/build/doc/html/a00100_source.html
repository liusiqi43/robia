333  template<typename GRAPH, typename VECTOR_OP>
334  class Tick {
335  VECTOR_OP& op;
336  double dt;
337 
338  public:
339  Tick(VECTOR_OP& vop, double delta_t)
340  :op(vop),dt(delta_t) {}
341 
342  bool operator()(typename GRAPH::vertex_type& n) {
343  NeighborsMeanSpeed<GRAPH,VECTOR_OP> mean(op,n);
344  switch(n.value.getStatus()) {
345  case statusNew:
346  n.for_each_edge(mean);
347  n.value.onNewSpeedUpdate(mean.get());
348  break;
349  case statusNoPreviousProto:
350  n.for_each_edge(mean);
351  n.value.onNoPrevSpeedUpdate(op,mean.get());
352  break;
353  case statusOk:
354  n.value.onSpeedUpdate(op,dt);
355  break;
356  default:
357  std::cerr << "Error : Tick : bad status. Report bug." << std::endl;
358  break;
359  }
360  return false; // the element should not be removed.
361  }
