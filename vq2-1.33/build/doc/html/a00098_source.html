44  template<typename ITER, typename VALUE_OF>
45  void shortest_confidence_interval(const ITER& begin,
46  const ITER& end,
47  const VALUE_OF& value_of,
48  double d,
49  double& min, double& max) {
50  unsigned int size = end - begin;
51 
52  if(size == 0)
53  return;
54  else if(size <= 2) {
55  min = value_of(*begin);
56  max = min;
57  }
58  else {
59 
60  std::sort(begin,end,vq2::functor::sort(value_of,*begin));
61 
62  unsigned int window_size = (unsigned int)(size*d+.5);
63  unsigned int upper_bound = size-window_size+1;
64  unsigned int kmin = 0;
65  double delta_min,delta;
66  delta_min = value_of(*(begin+(size-1)))-value_of(*begin)+1;
67  for(unsigned int k=0; k<upper_bound; ++k)
68  if((delta = value_of(*(begin+(k + (window_size-1)))) - value_of(*(begin+k))) < delta_min) {
69  delta_min = delta;
70  kmin = k;
71  }
72 
73  min = value_of(*(begin+kmin));
74  max = value_of(*(begin+(kmin + (window_size-1))));
75  }
76  }
