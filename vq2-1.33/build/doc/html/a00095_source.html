807  void computeConnectedComponents(std::map<unsigned int,Component*>& connected_components,
808  bool do_labelling) {
809  ClearTag clear_tag;
810  NewComponent newcomp(components);
811  typename std::vector<Component*>::iterator citer,cend;
812  unsigned int l;
813 
814  for_each_vertex(clear_tag);
815  clearComponents();
816  for_each(vertexes,newcomp);
817 
818  connected_components.clear();
819 
820  if(do_labelling) {
821  std::map<std::pair<unsigned int,unsigned int>,
822  unsigned int>::iterator occ;
823  // Elements are stored in descending order.
824  std::multimap<unsigned int,
825  std::pair<unsigned int,
826  unsigned int>,
827  invertorderop> secnerucco;
828  OccLabel occlabel;
829  unsigned int comp, label;
830  std::vector<unsigned int> labelling(components.size(),0);
831  typename std::vector<unsigned int>::iterator pliter,plend;
832 
833 
834  // Let us compute the map : [comp,label] -> nb : 'nb' edges of
835  // connected component 'comp' have label 'label'
836  for(citer = components.begin(),
837  cend = components.end(),
838  occlabel.comp = 0;
839  citer != cend;
840  ++citer,
841  occlabel.comp++)
842  (*citer)->for_each_edge(occlabel);
843 
844  // Now, we build the reverse map.
845  // nb -> (comp,label)
846  for(occ = occlabel.occurences.begin();
847  occ != occlabel.occurences.end();
848  ++occ)
849  secnerucco.insert(std::pair<unsigned int,
850  std::pair<unsigned int,
851  unsigned int > >(occ->second,occ->first));
852 
853  // Now, we can start real labelling, from main labels in the edges.
854  typename std::multimap<unsigned int, std::pair<unsigned int,unsigned int>, invertorderop>::iterator cco;
855  std::set<unsigned int> used_labels;
856  for(cco = secnerucco.begin(); cco != secnerucco.end(); cco++) {
857  comp = cco->second.first;
858  label = cco->second.second;
859  if(labelling[comp] == 0 && (used_labels.count(label)==0))
860  labelling[comp] = label;
861  used_labels.insert(label);
862  }
863 
864  // There may be some unlabeled components.
865  for(comp = 0,
866  pliter = labelling.begin(),
867  plend = labelling.end();
868  pliter != plend;
869  ++comp,++pliter)
870  if(*pliter == 0) {
871  *pliter = nextLabel();
872  }
873 
874 
875  LabelVertex label_vertex;
876  LabelEdge label_edge;
877  // Label all vertexes and edges in each component.
878  for(citer = components.begin(),
879  cend = components.end(),
880  pliter = labelling.begin();
881  citer != cend;
882  ++citer, ++pliter) {
883  label_vertex.label = *pliter;
884  label_edge.label = *pliter;
885  (*citer)->for_each_vertex(label_vertex);
886  (*citer)->for_each_edge(label_edge);
887  connected_components[*pliter]=*citer;
888  }
889 
890  }
891  else {
892  for(l = 1, citer = components.begin(), cend = components.end();
893  citer != cend;
894  ++citer,++l) {
895  connected_components[l]=*citer;
896  }
897  }
898  }
